-- Tạo 30 items và chèn dữ liệu vào bảng Inventory
DECLARE @i INT = 1;
WHILE @i <= 30
BEGIN
    INSERT INTO Inventory (InventoryID, ItemID, SizeID, Quantity, ImportDate, LocID)
    SELECT 'INV' + RIGHT('00' + CAST(@i AS NVARCHAR(2)), 2) AS InventoryID,
           'ITEM' + RIGHT('00' + CAST(@i AS NVARCHAR(2)), 2) AS ItemID,
           SizeID,
           FLOOR(RAND()*(20-5+1))+5 AS Quantity, -- Số lượng ngẫu nhiên từ 5 đến 20
           '2024-04-20' AS ImportDate, -- Ngày nhập cố định
           'LOC' + RIGHT('00' + CAST(FLOOR(RAND()*(4-1+1))+1 AS NVARCHAR(2)), 2) AS LocID -- Vị trí ngẫu nhiên từ LOC01 đến LOC04
    FROM (
        SELECT DISTINCT FLOOR(RAND()*(6-1+1))+1 AS SizeID -- Chọn kích cỡ ngẫu nhiên từ 1 đến 6 cho mỗi mục
        FROM Items
    ) AS ItemSizes;
    SET @i = @i + 1;
END;
